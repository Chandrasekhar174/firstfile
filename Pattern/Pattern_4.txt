import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n=sc.nextInt();
		for(int i=0;i<n;i++)
		{
		    int k=0;
		    for(k=0;k<n-i-1;k++)
		    {
		        System.out.print("  ");
		    }
		    for(int j=k;j<n;j++)
		    {
		        System.out.print("* ");
		    }
		    System.out.println();
		    
		}
	}
}
